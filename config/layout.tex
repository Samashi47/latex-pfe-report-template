% Page geometry
\geometry{
    top=2.5cm,
    bottom=2.5cm,
    left=2cm,
    right=2cm,
    headheight=2.5cm,
    head=1.25cm,
    foot=1.25cm
}

% Headers and footers

\renewcommand{\chaptermark}[1]{\markboth{Chapter \thechapter: #1}{}}
\fancypagestyle{mainmatter}{
    \fancyhf{} % Clear all header and footer fields
    \fancyhead[L]{\large\textbf{\leftmark}} % Chapter name in left header
    \fancyfoot[R]{\bfseries Page | \thepage}   % Page number in right footer
    \renewcommand{\headrulewidth}{0.2mm} % Header rule for all pages except first chapter page
}

\fancypagestyle{chapter-first-page}{
    \fancyhf{} % Clear header/footer
    \fancyfoot[R]{\bfseries Page | \thepage}
    \renewcommand{\headrulewidth}{0mm} % No header rule for first page of chapter
}

\makeatletter
\let\old@chapter\@chapter%
\def\@chapter[#1]#2{\old@chapter[#1]{#2}\thispagestyle{chapter-first-page}}
\let\old@schapter\@schapter%
\def\@schapter#1{\old@schapter{#1}\markboth{#1}{#1}\thispagestyle{chapter-first-page}}
\makeatother
